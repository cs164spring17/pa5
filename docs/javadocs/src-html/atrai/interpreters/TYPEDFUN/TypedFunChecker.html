022class PrimitiveTypeValue extends TypeValue {
023    public static final PrimitiveTypeValue INT = new PrimitiveTypeValue(PrimitiveTypeValue.TypeName.INT);
024    public static final PrimitiveTypeValue STRING = new PrimitiveTypeValue(PrimitiveTypeValue.TypeName.STRING);
025    public static final PrimitiveTypeValue BOOL = new PrimitiveTypeValue(PrimitiveTypeValue.TypeName.BOOL);
026    public static final PrimitiveTypeValue NULL = new PrimitiveTypeValue(PrimitiveTypeValue.TypeName.NULL);
027    private final PrimitiveTypeValue.TypeName val;
028
029    private PrimitiveTypeValue(PrimitiveTypeValue.TypeName val) {
030        this.val = val;
031    }
032
033    @Override
034    public boolean equals(Object obj) {
035        return obj == this;
036    }
037
038    @Override
039    public String toString() {
040        return val.toString();
041    }
042
043    enum TypeName {INT, STRING, BOOL, NULL}
044}
045
046class FunctionTypeValue extends TypeValue {
047    private TypeValue argType = null;
048    private TypeValue returnType;
049
050    public FunctionTypeValue(TypeValue argType, TypeValue returnType) {
051        this.argType = argType;
052        this.returnType = returnType;
053    }
054
055    public TypeValue getReturnType() {
056        return returnType;
057    }
058
059    public TypeValue getArgType() {
060        return argType;
061    }
062
063    @Override
064    public boolean equals(Object obj) {
065        if (!(obj instanceof FunctionTypeValue)) {
066            return false;
067        }
068        FunctionTypeValue other = (FunctionTypeValue) obj;
069        return Objects.equals(argType, other.argType) && Objects.equals(returnType, other.returnType);
070    }
071
072    @Override
073    public String toString() {
074        StringBuilder sb = new StringBuilder();
075        sb.append('(');
076        if (argType != null) sb.append(argType);
077        sb.append(")->").append(returnType);
078        return sb.toString();
079    }
080}
